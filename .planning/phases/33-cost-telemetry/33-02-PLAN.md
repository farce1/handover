    ```typescript
    /**
     * `handover cost` — read-only CLI subcommand displaying cost telemetry.
     *
     * Two output modes: text (default, scannable in 3 seconds) and JSON (--json,
     * pinned formatVersion: 1 for Phase 36 GitHub Action consumption — D-11).
     *
     * Mirrors src/cli/reindex.ts overall shape and src/regen/dep-graph.ts formatDryRun /
     * formatDryRunJson dual-format precedent.
     *
     * Failure modes (D-21):
     *   - missing DB:    'No telemetry yet.' + exit 0
     *   - corrupt DB:    HandoverError + exit 1 via handleCliError
     *   - bad --since-date: exit 2 via HandoverError (Commander 'bad input')
     */
    import pc from 'picocolors';
    import { existsSync } from 'node:fs';
    import { join, resolve } from 'node:path';
    import { handleCliError, HandoverError } from '../utils/errors.js';
    import {
      openTelemetryDb,
      getRecentRuns,
      getRendererSummary,
      TELEMETRY_VERSION,
      type RecentRunRow,
      type RendererSummaryRow,
    } from '../regen/telemetry/index.js';

    export interface CostCommandOptions {
      runs?: number;
      sinceDate?: string;
      renderer?: string;
      view?: 'attributed' | 'standalone';
      json?: boolean;
    }

    /** Format-version envelope shape for --json output (D-11). Phase 36 pins to formatVersion 1. */
    interface CostJsonPayload {
      formatVersion: 1;
      telemetryVersion: typeof TELEMETRY_VERSION;
      window: { kind: 'runs'; count: number } | { kind: 'since-date'; sinceIso: string };
      view: 'attributed' | 'standalone';
      runs: Array<{
        runId: string;
        startedAt: string;
        elapsedMs: number;
        totalCostUsd: number;
        totalInputTokens: number;
        totalOutputTokens: number;
        topModel: string;
        provider: string;
        thresholdUsd: number | null;
        thresholdExceeded: boolean;
        sinceRef: string | null;
      }>;
      rendererAggregate: Array<{
        rendererId: string;
        avgElapsedMs: number;
        totalAttributedCostUsd: number;
        totalInputTokens: number;
        totalOutputTokens: number;
        runsCount: number;
      }>;
    }

    export async function runCost(opts: CostCommandOptions): Promise<void> {
      try {
        // 1. Validate flag combinations.
        if (opts.runs != null && opts.sinceDate != null) {
          throw new HandoverError(
            'Conflicting flags',
            '--runs and --since-date are mutually exclusive (D-10).',
            'Use either --runs <N> or --since-date <ISO>, not both.',
            'COST_FLAGS_MUTEX',
          );
        }
        if (opts.sinceDate != null) {
          const parsed = Date.parse(opts.sinceDate);
          if (Number.isNaN(parsed)) {
            throw new HandoverError(
              'Invalid --since-date',
              `'${opts.sinceDate}' is not a valid ISO-8601 date.`,
              'Use ISO format like 2026-04-01 or 2026-04-01T00:00:00Z.',
              'COST_BAD_SINCE_DATE',
            );
          }
        }
        const view: 'attributed' | 'standalone' = opts.view ?? 'attributed';
        if (view !== 'attributed' && view !== 'standalone') {
          throw new HandoverError(
            'Invalid --view',
            `'${view}' is not a recognized view.`,
            'Use --view attributed (default) or --view standalone.',
            'COST_BAD_VIEW',
          );
        }

        // 2. Resolve DB path.
        const rootDir = resolve(process.cwd());
        const dbPath = join(rootDir, '.handover', 'telemetry.db');

        // 3. Empty-DB path (D-21).
        if (!existsSync(dbPath)) {
          if (opts.json) {
            const empty: CostJsonPayload = {
              formatVersion: 1,
              telemetryVersion: TELEMETRY_VERSION,
              window: { kind: 'runs', count: opts.runs ?? 10 },
              view,
              runs: [],
              rendererAggregate: [],
            };
            process.stdout.write(JSON.stringify(empty, null, 2) + '\n');
          } else {
            process.stdout.write('No telemetry yet. Run `handover generate` to record cost data.\n');
          }
          return;
        }

        // 4. Open DB. Corrupt-DB throws → HandoverError (D-21).
        let db: ReturnType<typeof openTelemetryDb>;
        try {
          db = openTelemetryDb(dbPath);
        } catch (err) {
          throw new HandoverError(
            'Telemetry DB unreadable',
            `SQLite open failed: ${(err as Error).message}`,
            'Delete .handover/telemetry.db to reset (history will be lost; this is local metadata, never committed).',
            'TELEMETRY_DB_CORRUPT',
          );
        }

        try {
          // 5. Build query options.
          const sinceIso = opts.sinceDate ? new Date(opts.sinceDate).toISOString() : undefined;
          const limit = opts.runs ?? 10;
          const recentRuns = getRecentRuns(db, { limit, sinceIso });
          const rendererAggregate = getRendererSummary(db, {
            limit, sinceIso, rendererId: opts.renderer, view,
          });

          // 6. Emit.
          if (opts.json) {
            const payload: CostJsonPayload = {
              formatVersion: 1,
              telemetryVersion: TELEMETRY_VERSION,
              window: sinceIso
                ? { kind: 'since-date', sinceIso }
                : { kind: 'runs', count: limit },
              view,
              runs: recentRuns.map((r) => ({
                runId: r.runId,
                startedAt: r.startedAt,
                elapsedMs: r.elapsedMs,
                totalCostUsd: r.totalCostUsd,
                totalInputTokens: r.totalInputTokens,
                totalOutputTokens: r.totalOutputTokens,
                topModel: r.topModel,
                provider: r.provider,
                thresholdUsd: r.thresholdUsd,
                thresholdExceeded: r.thresholdExceeded,
                sinceRef: r.sinceRef,
              })),
              rendererAggregate: rendererAggregate.map((r) => ({
                rendererId: r.rendererId,
                avgElapsedMs: r.avgElapsedMs,
                totalAttributedCostUsd: r.totalAttributedCostUsd,
                totalInputTokens: r.totalInputTokens,
                totalOutputTokens: r.totalOutputTokens,
                runsCount: r.runsCount,
              })),
            };
            process.stdout.write(JSON.stringify(payload, null, 2) + '\n');
          } else {
            process.stdout.write(formatText(recentRuns, rendererAggregate, limit, view));
          }
        } finally {
          db.close();
        }
      } catch (err) {
        handleCliError(err, 'handover cost');
      }
    }

    /**
     * Two-section scannable text view (D-09 minimal-chrome).
     * No box-drawing. ⚠ marker for over-threshold rows.
     */
    function formatText(
      runs: RecentRunRow[],
      rendererAggregate: RendererSummaryRow[],
      limit: number,
      view: 'attributed' | 'standalone',
    ): string {
      const lines: string[] = [];
      lines.push(`Recent runs (last ${runs.length} of ${limit}):`);
      lines.push('  run_id                                when                   cost      tokens (in/out)        top model           flags');
      for (const r of runs) {
        const shortId = r.runId.slice(0, 8);
        const when = r.startedAt;
        const cost = `$${r.totalCostUsd.toFixed(4)}`;
        const tokens = `${formatTokensCompact(r.totalInputTokens)} / ${formatTokensCompact(r.totalOutputTokens)}`;
        const flag = r.thresholdExceeded ? pc.yellow('⚠ over threshold') : '';
        lines.push(`  ${shortId}  ${when}  ${cost.padEnd(8)}  ${tokens.padEnd(20)}  ${r.topModel.padEnd(18)}  ${flag}`);
      }
      lines.push('');
      const viewLabel = view === 'standalone'
        ? `Per-renderer standalone (no division across consumers; totals double-count)`
        : `Per-renderer aggregate (attributed cost — Σ equals Σ round_runs.cost_usd)`;
      lines.push(`${viewLabel} (over ${runs.length} runs):`);
      lines.push('  renderer                  avg time   total cost   tokens (in/out)        runs');
      for (const r of rendererAggregate) {
        const avgTime = `${(r.avgElapsedMs / 1000).toFixed(1)}s`;
        const cost = `$${r.totalAttributedCostUsd.toFixed(4)}`;
        const tokens = `${formatTokensCompact(r.totalInputTokens)} / ${formatTokensCompact(r.totalOutputTokens)}`;
        lines.push(`  ${r.rendererId.padEnd(24)}  ${avgTime.padEnd(8)}  ${cost.padEnd(10)}  ${tokens.padEnd(20)}  ${r.runsCount}`);
      }
      lines.push('');
      return lines.join('\n');
    }

    function formatTokensCompact(n: number): string {
      if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(1)}M`;
      if (n >= 1_000)    return `${(n / 1_000).toFixed(1)}k`;
      return `${n}`;
    }
    ```
